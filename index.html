  function plantIcon(type) {

    const t = (type||'').toLowerCase();

    if (t.includes('tree'))      return '🌳';

    if (t.includes('shrub'))     return '🌿';

    if (t.includes('annual'))    return '🌼';

    if (t.includes('perennial')) return '🌺';

    if (t.includes('herb'))      return '🌱';

    if (t.includes('vine'))      return '🍃';

    if (t.includes('grass'))     return '🌾';

    return '🪴';

  }


  function updateCount(shown) {

    const total = Object.keys(plants).length;

    document.getElementById('legend-count').textContent = total;

    const f = document.getElementById('legend-filter');

    if (shown !== undefined && shown !== total) {

      f.textContent = shown + ' shown';

      f.classList.add('visible');

    } else {

      f.classList.remove('visible');

    }

  }


  // ── Pin ──

  function createPinEl(id, data) {

    const map = document.getElementById('map-container');

    const div = document.createElement('div');

    div.className   = 'pin';

    div.dataset.id  = id;

    div.style.left  = (data.px * 100) + '%';

    div.style.top   = (data.py * 100) + '%';

    div.innerHTML = `

      <svg width="28" height="36" viewBox="0 0 28 36" fill="none" xmlns="http://www.w3.org/2000/svg">

        <path d="M14 0C6.268 0 0 6.268 0 14c0 9.333 14 22 14 22S28 23.333 28 14C28 6.268 21.732 0 14 0z" fill="#e85c3a"/>

        <circle cx="14" cy="14" r="6" fill="white" opacity="0.92"/>

        <circle cx="14" cy="14" r="3.5" fill="#e85c3a"/>

      </svg>

      <span class="pin-label">${escapeHtml(data.name||'Plant')}</span>`;

    div.addEventListener('click', e => { e.stopPropagation(); openViewPanel(id); });

    map.appendChild(div);

    return div;

  }


  // ── Load ──

  async function loadPlants() {

    try {

      const snap = await getDocs(collection(db, COL));

      snap.forEach(d => {

        const data = d.data();

        const el = createPinEl(d.id, data);

        plants[d.id] = { data, el };

      });

    } catch(e) {

      console.error(e);

      toast('Could not load plants. Check Firebase config.');

    }

    document.getElementById('loading').classList.add('hidden');

    updateCount();

  }


  // ── Map click ──

  document.getElementById('map-container').addEventListener('click', e => {

    if (e.target.closest('.pin')) return;

    closeSearch();

    const rect = e.currentTarget.getBoundingClientRect();

    pendingX = (e.clientX - rect.left) / rect.width;

    pendingY = (e.clientY - rect.top)  / rect.height;

    openAddForm();

  });


  // ── Add form ──

  function openAddForm() {

    editingId = null;

    document.getElementById('form-title').textContent    = 'New Plant';

    document.getElementById('form-subtitle').textContent = 'Fill in what you know';

    document.getElementById('form-save').textContent     = 'Save Plant';

    ['f-name','f-type','f-bloom','f-water','f-fert','f-notes']

      .forEach(id => document.getElementById(id).value = '');

    showOverlay('form-overlay');

    setTimeout(() => document.getElementById('f-name').focus(), 380);

  }


  // ── Edit form ──

  function openEditForm(id) {

    const d = plants[id].data;

    editingId = id;

    document.getElementById('form-title').textContent    = 'Edit Plant';

    document.getElementById('form-subtitle').textContent = d.name || '';

    document.getElementById('form-save').textContent     = 'Update Plant';

    document.getElementById('f-name').value  = d.name  || '';

    document.getElementById('f-type').value  = d.type  || '';

    document.getElementById('f-bloom').value = d.bloom || '';

    document.getElementById('f-water').value = d.water || '';

    document.getElementById('f-fert').value  = d.fert  || '';

    document.getElementById('f-notes').value = d.notes || '';

    hideOverlay('view-overlay');

    showOverlay('form-overlay');

    setTimeout(() => document.getElementById('f-name').focus(), 380);

  }


  document.getElementById('form-close').addEventListener('click',  () => hideOverlay('form-overlay'));

  document.getElementById('form-cancel').addEventListener('click', () => hideOverlay('form-overlay'));


  document.getElementById('form-save').addEventListener('click', async () => {

    const name = val('f-name');

    if (!name) { toast('Please enter a plant name.'); document.getElementById('f-name').focus(); return; }


    const data = {

      name,

      type:  val('f-type'),

      bloom: val('f-bloom'),

      water: val('f-water'),

      fert:  val('f-fert'),

      notes: val('f-notes'),

    };


    const btn = document.getElementById('form-save');

    btn.textContent = '…';

    btn.disabled = true;


    try {

      if (editingId) {

        data.px = plants[editingId].data.px;

        data.py = plants[editingId].data.py;

        await updateDoc(doc(db, COL, editingId), data);

        plants[editingId].data = data;

        const lbl = plants[editingId].el.querySelector('.pin-label');

        if (lbl) lbl.textContent = data.name;

        hideOverlay('form-overlay');

        toast('Plant updated ✓');

      } else {

        data.px = pendingX;

        data.py = pendingY;

        const ref = await addDoc(collection(db, COL), data);

        const el  = createPinEl(ref.id, data);

        plants[ref.id] = { data, el };

        hideOverlay('form-overlay');

        toast('Plant added ✓');

      }

      updateCount();

    } catch(e) {

      console.error(e);

      toast('Error saving. Please try again.');

    }


    btn.disabled    = false;

    btn.textContent = editingId ? 'Update Plant' : 'Save Plant';

  });


  // ── View panel ──

  function openViewPanel(id) {

    viewingId = id;

    const d = plants[id].data;

    setText('v-name',     d.name);

    setText('v-type-sub', d.type || 'Plant');

    document.getElementById('v-badge').textContent = d.type || 'Plant';

    setText('v-water', d.water);

    setText('v-bloom', d.bloom);

    setText('v-fert',  d.fert);


    const notesBlock = document.getElementById('v-notes-block');

    if (d.notes && d.notes.trim()) {

      document.getElementById('v-notes').textContent = d.notes;

      notesBlock.style.display = '';

    } else {

      notesBlock.style.display = 'none';

    }

    showOverlay('view-overlay');

  }


  document.getElementById('view-close').addEventListener('click', () => {

    viewingId = null; hideOverlay('view-overlay');

  });

  document.getElementById('v-done').addEventListener('click', () => {

    viewingId = null; hideOverlay('view-overlay');

  });


  document.getElementById('v-edit').addEventListener('click', () => {

    if (viewingId && plants[viewingId]) openEditForm(viewingId);

  });


  document.getElementById('v-delete').addEventListener('click', () => {

    if (!viewingId) return;

    pendingDeleteId = viewingId;

    document.getElementById('confirm-name').textContent =

      plants[pendingDeleteId]?.data?.name || 'this plant';

    document.getElementById('confirm-overlay').classList.add('visible');

  });


  document.getElementById('confirm-cancel').addEventListener('click', () => {

    pendingDeleteId = null;

    document.getElementById('confirm-overlay').classList.remove('visible');

  });


  document.getElementById('confirm-yes').addEventListener('click', async () => {

    if (!pendingDeleteId) return;

    const id = pendingDeleteId;

    pendingDeleteId = null;

    document.getElementById('confirm-overlay').classList.remove('visible');

    try {

      await deleteDoc(doc(db, COL, id));

      plants[id].el.remove();

      delete plants[id];

      viewingId = null;

      hideOverlay('view-overlay');

      toast('Plant removed.');

      updateCount();

    } catch(e) {

      console.error(e);

      toast('Error deleting. Please try again.');

    }

  });


  ['form-overlay','view-overlay'].forEach(oid => {

    document.getElementById(oid).addEventListener('click', e => {

      if (e.target !== e.currentTarget) return;

      if (oid === 'view-overlay') viewingId = null;

      hideOverlay(oid);

    });

  });


  // ── Search ──

  const searchInput   = document.getElementById('search-input');

  const searchClear   = document.getElementById('search-clear');

  const searchResults = document.getElementById('search-results');


  function closeSearch() {

    searchInput.value = '';

    searchClear.classList.remove('visible');

    searchResults.classList.remove('visible');

    searchResults.innerHTML = '';

    Object.values(plants).forEach(({ el }) => el.classList.remove('faded','highlighted'));

    updateCount();

  }


  function doSearch(query) {

    searchResults.innerHTML = '';

    const q = query.toLowerCase().trim();


    if (!q) { closeSearch(); return; }


    searchClear.classList.add('visible');


    const matches = Object.entries(plants).filter(([, {data}]) =>

      (data.name  || '').toLowerCase().includes(q) ||

      (data.type  || '').toLowerCase().includes(q) ||

      (data.bloom || '').toLowerCase().includes(q)

    );


    Object.entries(plants).forEach(([id, {el}]) => {

      const hit = matches.some(([mid]) => mid === id);

      el.classList.toggle('faded',       !hit);

      el.classList.toggle('highlighted',  hit);

    });


    updateCount(matches.length);


    if (matches.length === 0) {

      searchResults.innerHTML = `<div class="result-empty">No plants match "${escapeHtml(query)}"</div>`;

    } else {

      matches.forEach(([id, {data}]) => {

        const item = document.createElement('div');

        item.className = 'result-item';

        item.innerHTML = `

          <div class="result-icon">${plantIcon(data.type)}</div>

          <div class="result-info">

            <div class="result-name">${escapeHtml(data.name)}</div>

            <div class="result-type">${escapeHtml(data.type||'Plant')}</div>

          </div>

          <div class="result-arrow">›</div>`;

        item.addEventListener('click', () => { closeSearch(); openViewPanel(id); });

        searchResults.appendChild(item);

      });

    }

    searchResults.classList.add('visible');

  }


  searchInput.addEventListener('input', e => doSearch(e.target.value));

  searchInput.addEventListener('focus', e => { if (e.target.value) doSearch(e.target.value); });

  searchClear.addEventListener('click', closeSearch);

  document.addEventListener('keydown', e => { if (e.key === 'Escape') closeSearch(); });


  // ── Init ──

  loadPlants();
